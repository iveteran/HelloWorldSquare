Get-NetConnectionProfile
New-NetFirewallRule -Name 'OpenSSH-Server-In-TCP' -DisplayName 'OpenSSH Server (sshd)' -Enabled True -Direction Inbound -Protocol TCP -Action Allow -LocalPort 22
Set-NetFirewallRule -Name "OpenSSH-Server-In-TCP" -Profile Any
get-netfirewallrule -Name "OpenSSH-Server-In-TCP"

# 查看所有配置文件的防火墙状态
netsh advfirewall show allprofiles
# 查看当前配置文件状态
netsh advfirewall show currentprofile

# 查看所有入站规则
netsh advfirewall firewall show rule name=all dir=in
# 按名称搜索规则
netsh advfirewall firewall show rule name="规则名称"
# 查看某个端口的规则
netsh advfirewall firewall show rule name=all dir=in | findstr "8080"

# 开放 TCP 端口（以 8080 为例）
netsh advfirewall firewall add rule name="Open Port 8080" dir=in action=allow protocol=TCP localport=8080
# 开放 UDP 端口
netsh advfirewall firewall add rule name="Open UDP 8080" dir=in action=allow protocol=UDP localport=8080
# 同时开放 TCP + UDP
netsh advfirewall firewall add rule name="Open Port 8080" dir=in action=allow protocol=ANY localport=8080

# 例：开放Clash端口
netsh advfirewall firewall add rule name="Open Port Clash" dir=in action=allow protocol=TCP localport=7890

# 删除规则
netsh advfirewall firewall delete rule name="Open Port 8080"
# 禁用某条规则（不删除）
netsh advfirewall firewall set rule name="Open Port 8080" new enable=no
# 启用某条规则
netsh advfirewall firewall set rule name="Open Port 8080" new enable=yes

# 关闭所有配置文件防火墙（慎用）
netsh advfirewall set allprofiles state off
# 开启防火墙
netsh advfirewall set allprofiles state on
